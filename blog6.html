My Arduino Robotics Projects

Push Button

int BUTTON_PIN = 2; void setup() { pinMode(BUTTON_PIN, INPUT); Serial.begin(9600); } void loop() { if (digitalRead(BUTTON_PIN) == HIGH) { Serial.println("BUTTON IS PRESSED"); } else { Serial.println("BUTTON IS NOT PRESSED"); } delay(100); }

Controlling LED Brightness with Potentiometer

int potPin = A0; int ledPin = 9; void setup() { pinMode(ledPin, OUTPUT); } void loop() { int potValue = analogRead(potPin); int ledBrightness = map(potValue, 0, 1023, 0, 255); analogWrite(ledPin, ledBrightness); delay(1000); }

Passive Buzzer Melody

int buzzer = 13; void setup() { pinMode(buzzer, OUTPUT); } void loop() { tone(buzzer, random(500, 1500)); delay(300); }

Active Buzzer Alarm

int buzzer = 13; void setup() { pinMode(buzzer, OUTPUT); } void loop() { digitalWrite(buzzer, HIGH); delay(500); digitalWrite(buzzer, LOW); delay(500); }
